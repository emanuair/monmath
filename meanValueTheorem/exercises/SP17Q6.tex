\outcome{Find the linear approximation to a function at a point and use it to approximate the function value.}
\outcome{Use the Mean Value Theorem to bound the error in linear approximation.}

Let $f$ be a differentiable function on the interval $(0,6)$ with the particular values of $f$ and $f'$ given in the table below.
\[
\begin{array}{c|c|c|}
x & f(x) & f'(x)\\ \hline
1 &  0 &     4\\
2 &  1 &     3\\
3 &  5 &     2\\
4 &  6 &     5\\
\end{array}
\]

\begin{exercise}
Find the average rate of change of $f$ on the interval $[1,4]$.
\[
\text{average rate of change of $f$ on $[1,4]$}=\answer{2}
\]
\begin{exercise}
Choose the correct statement.
\begin{multipleChoice}
\choice{There exists a point $c$ in $(1,4)$ such that the instantaneous rate of change of $f$ at the point $c$ is equal to the average rate of change of $f$ on $[1,4]$ , although the function $f$ does not satisfy the conditions of the Mean value theorem on $[1,4]$.}
\choice[correct]{There exists a point $c$ in $(1,4)$  such that the instantaneous rate of change of $f$ at the point $c$ is equal to the average rate of change of $f$ on $[1,4]$, since the function $f$ satisfies the conditions of the Mean value theorem on $[1,4]$.}
\choice{There exists a point $c$ in $(1,4)$ such that the instantaneous rate of change of $f$ at the point $c$ is equal to the average rate of change of $f$ on $[1,4]$, but we don't have enough information to find such a point.}
\choice{There exists no point $c$ in $(1,4)$ such that the instantaneous rate of change of $f$ at the point $c$ is equal to the average rate of change of $f$ on $[1,4]$.}
\end{multipleChoice}

\end{exercise}
\end{exercise}
