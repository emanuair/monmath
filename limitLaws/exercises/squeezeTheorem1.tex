\author{Nela Lakos \and Kyle Parsons}

\begin{exercise}

Given a function $f$ such that for all $x>0$, 
\[
\cos x \leq f(x) \leq e^x.
\]

Then $\lim_{x\to0^+}f(x) = \answer{1}$.

\begin{hint}
Use the Squeeze Theorem.
\end{hint}

\end{exercise}
