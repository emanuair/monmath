\begin{exercise}
Let $h$ be a function defined on $(0,2)$ such that $f(x)\le h(x)\le
g(x)$ for all $x$ with $0<x<2$ except possibly at $x=1$. Then
\[
\lim_{x\to 1}h(x)= \answer{0}
\]
due to \wordChoice{\choice{limit laws}\choice{continuity}\choice{of the difference
    law}\choice{quotient law}\choice[correct]{the Squeeze Theorem}}.
\begin{exercise}
Compute the limit.
\[ 
\lim_{x\to 1}\frac{e^{h(x)}}{\cos{(g(x))}}=\frac{e^{\answer{0}}}{\cos\left(\answer{0}\right)}=\answer{1}
\]  

\end{exercise}
\end{exercise}
\end{exercise}
