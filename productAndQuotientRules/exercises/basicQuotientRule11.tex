\begin{exercise}
Let $f(x) = \frac{ 2 e^x-\sqrt[5]{x}}{-3 \sqrt[5]{x}}$. Compute:
\[
f'(x)
\begin{prompt}=\answer{\frac{2 e^x-{x}^{\frac{1}{5}}}{15 {x}^\frac{6}{5}}-\frac{2 e^x-\frac{1}{5 {x}^\frac{4}{5}}}{3 {x}^\frac{1}{5}}}\end{prompt}
\]
\end{exercise}
