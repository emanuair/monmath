\begin{exercise}
Let $R(k) = -\frac{2 \left(\frac{1}{k^4}+3 \sqrt{k}\right)}{k^5}$. Compute:
\[
R'(k)
\begin{prompt}=\answer{\frac{10 \left(\frac{1}{k^4}+3 \sqrt{k}\right)}{k^6}-\frac{2 \left(\frac{3}{2 \sqrt{k}}-\frac{4}{k^5}\right)}{k^5}}\end{prompt}
\]
\end{exercise}
