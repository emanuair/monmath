\begin{exercise}

Consider the relation $x^3+y^2 = 1$.  Which of the following are valid expressions for $\frac{dy}{dx}$?
\begin{selectAll}
\choice[correct]{$\frac{-3x^2}{2y}$}
\choice{$\frac{-2y}{3x^2}$}
\choice[correct]{$\frac{-3x^2}{2y}+x^3+y^2-1$}
\choice{$\frac{-2y}{3x^2}+xy(x^3+y^2-1)$}
\end{selectAll}
\end{exercise}
