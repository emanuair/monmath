\begin{document}
\begin{exercise}

Use implicit differentiation to find $\frac{dy}{dx}$ as a function of $x$ and $y$ where
\[
x^3+y^3 = 9.
\]
\begin{prompt}
\[
\frac{dy}{dx} = \frac{-x^2}{y^2}
\]
\end{prompt}
What is the equation of the tangent line to this curve at the point $(1,2)$?
\begin{prompt}
\[
y = \answer{\frac{-1}{4}(x-1)+2}
\]
\end{prompt}
\end{exercise}
