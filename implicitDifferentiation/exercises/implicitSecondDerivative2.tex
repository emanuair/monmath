\begin{exercise}

Consider
\[
x^2+y^2 = 1.
\]
Compute
\[
\frac{dy}{dx} \begin{prompt}= \answer{-\frac{x}{y}}
\end{prompt}.
\]
Select all that apply:
\begin{selectAll}
	\choice[correct]{$\frac{d^2y}{dx^2} = -\frac{x^2+y^2}{y^3}$}
	\choice{$\frac{d^2y}{dx^2} = \frac{x^2-y^2}{y^3}$}
	\choice{$\frac{d^2y}{dx^2} = \frac{x^2+y^2}{y^4}$}
	\choice[correct]{$\frac{d^2y}{dx^2} = -\frac{1}{y^3}$}
\end{selectAll}
\end{exercise}
