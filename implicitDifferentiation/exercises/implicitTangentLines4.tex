\begin{document}
\begin{exercise}

Use implicit differentiation to find $\frac{dy}{dx}$ as a function of $x$ and $y$ where
\[
\cos(y^2) = x+y.
\]
\begin{prompt}
\[
\frac{dy}{dx} = \answer{\frac{-1}{1+2y\sin(y^2)}}
\]
\end{prompt}
What is the equation of the tangent line to this curve at the point $(1,0)$?
\begin{prompt}
\[
y = \answer{-1(x-1)}
\]
\end{prompt}
\end{exercise}
