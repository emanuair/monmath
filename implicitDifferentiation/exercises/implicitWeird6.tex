\begin{exercise}

Consider the relation 
\[
x^2+y^2 = 0.
\]
Which of the following statements is the most accurate?
\begin{multipleChoice}
	\choice{$\frac{dy}{dx} = -\frac{x}{y}$}
	\choice[correct]{There are no functions implicitly defined by this relation, so $\frac{dy}{dx}$ is meaningless}
	\choice{$\frac{dy}{dx} = -\frac{x}{y}$, but the only point satisfying the relation is $(0,0)$, so it is undefined}
\end{multipleChoice}
\end{exercise}
