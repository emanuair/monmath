\begin{problem}
  Let:
  \begin{align*}
    R &= \{(x,y): x^2 +y^2 \le 1~\text{with}~x\ge0, y\ge0\}\\
    T &= \{(x,y): x+y\le 1~\text{with}~x\ge0, y\ge0\}
  \end{align*}
  Additionally, let $F(x,y) = \sqrt{1-x^2-y^2}$ and $G(x,y) = 1-x-y$.
  Which quantity is largest?
  \begin{multipleChoice}
    \pdfOnly{\begin{multicols}{2}}
      \choice[correct]{$\iint_R  \sqrt{1+F^{(1,0)}(x,y)^2+F^{(0,1)}(x,y)^2} \d A$}
      \choice{$\iint_T \sqrt{1+G^{(1,0)}(x,y)^2+G^{(0,1)}(x,y)^2} \d A$}
      \pdfOnly{\end{multicols}}
  \end{multipleChoice}
  \end{problem}
