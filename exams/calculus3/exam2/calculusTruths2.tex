\begin{exercise}
  Which statements are necessarily true?
  \begin{selectAll}
    \choice{$\int_0^1\int_0^x F(x,y) \d y \d x= \int_0^1\int_y^1
      F(x,y) \d y \d x$.}

    \choice[correct]{The integral
      $\int_{-1}^1\int_{-\sqrt{1-x^2}}^{\sqrt{1-x^2}} \d y \d x$ is
      the area of the unit disk.}

    \choice{$\int_0^1\int_x^1 F(x,y) \d x\d y = \int_0^1\int_y^1 F(x,y) \d y \d x$}

    \choice{Let $R=\{(x,y): x^2+y^2 \le 1\}$. If $\iint_R F(x,y) \d A
      =0$, then $F(x,y)=0$ for all $(x,y)$ in $R$.}

    \choice[correct]{If $F(x,y)= 10$, $\iint_R F(x,y)\d A$ is $10$ times the
      area of $R$.}
  \end{selectAll}
  \end{exercise}
