\begin{problem}%\points{6}
Consider the vector-valued function:
\[
\vec{b}(t) = \vector{5\cos(\pi t^2/2),5\sin(\pi t^2/2)} \quad\text{for }0\le t\le 2
\]
\textbf{Parameterize this function by arc length}. That is, find a new
vector-valued function $\vec{c}(s)$ that draws the same curve as
$\vec{b}(t)$ (as $t$ runs from $0$ to $2$) where $s$ is an \textbf{arc length
parameter.} Be sure to \textbf{give bounds} for $s$.

\begin{prompt}
  \[
  \vec{c}(s) = \vector{\answer{5\cos(s/5)},\answer{5\sin(s/5)}} \quad\text{for~}\answer{0}\le s\le \answer{10\pi}
  \]
\end{prompt}


\vspace{2.75in}

\end{problem}

\hrule

\begin{problem}
  Let $\vec{f}(t) = \vector{\cos(t),0,\sin(t)}$
  \begin{prompt}
    Select all true statements:
  \end{prompt}
  \begin{selectAll}
    \choice[correct]{$\vector{1,3,1}+t\cdot \vec{f}(\pi)$ is parameterized
      by arc length.}
    \choice{$\vector{2,1,-1}+t\cdot \vec{f}(t)$ is parameterized
      by arc length.}
    \choice{$\vector{\frac{1}{\sqrt{3}},\frac{1}{\sqrt{3}},\frac{1}{\sqrt{3}}}+t\cdot \vec{f}'(t)$ is parameterized
      by arc length.}
    \choice[correct]{$\vector{2,2,-1}+\vec{f}(t)$ is parameterized
      by arc length.}
  \end{selectAll}

  \end{problem}

\begin{problem}
  Let $c$ be a constant scalar, $\vec{v}$ and $\vec{w}$ be nonzero
  vectors and let $\vec{f}$ be a differentiable vector-valued
  function.
  \begin{prompt}
    Select statements that must be true:
  \end{prompt}
  \begin{selectAll}
    \choice{If $|\vec{f}(t)| = 2$ and $|\vec{f}'(t)| = 3$, then $|\vec{f}(t) \dotp \vec{f}'(t)| = 6$.}
    \choice[correct]{If $|\vec{f}(t)| = 2$ and $|\vec{f}'(t)| = 3$, then $|\vec{f}(t) \cross \vec{f}'(t)| = 6$.}
    \choice[correct]{$\dd{t} \left(c \vec{f}(t)\right) = c\dd{t} \vec{f}(t)$.}
    \choice{$\left|\dd{t} \vec{f}(t) \right| = \dd{t} | \vec{f}(t)|$.}
    \choice{$|\vec{v} + \vec{w}| = |\vec{v}| + |\vec{w}|$.}
  \end{selectAll}
\end{problem}
