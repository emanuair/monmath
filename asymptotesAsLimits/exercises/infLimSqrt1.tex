\begin{document}
\begin{exercise}
Let 
\[
R(n) = \frac{\sqrt{n^4}-n+2}{3 n-4 n^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{n\to \infty} R(n) \begin{prompt} = \answer{-\frac{1}{4}}\end{prompt}$
\item $\lim_{n\to -\infty}R(n) \begin{prompt} = \answer{-\frac{1}{4}}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{n^2}}{\frac{1}{n^2}}
\]
\end{hint}
\end{exercise}
