\begin{exercise}
Consider 
\[
f(\theta) = \frac{\theta ^2+7 \theta +12}{(\theta +1)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{\theta\to -1^- } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{\theta\to -1^+ } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{\theta\to -1 } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
