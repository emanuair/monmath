\begin{exercise}

The graph of a function $f$ is given in the figure below.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-10.3,xmax=10.3,ymin=-10.3,ymax=10.3,
        clip=true,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-10,-8,...,10},
	xtick={-10,-8,...,10},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \draw[very thick,red,dashed] (axis cs:0,-6) -- (axis cs:10.3,-6);
      \addplot[very thick,penColor,domain = -10.3:0,samples=50] plot{sin(360*x/pi) - x};
      \addplot[very thick,penColor,domain = 0:3,samples=50] plot{2*(x-3)/((3.3-x)*x)};
      \addplot[very thick,penColor,domain = 3:5,samples=50] plot{sqrt(2*x-6)};
      \addplot[very thick,penColor,domain = 5:10.3,samples=50] plot{12/((x-4)^2+2)-6};
      
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(0,0)};
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(5,2)};

      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(5,-2)};
      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(3,0)};

      \node[penColor] at (axis cs:3, 7) [penColor] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

Find the following values.  Possible answers are any number, $\infty$, $-\infty$, and $DNE$.

\begin{align*}
\lim_{x\to0^-}f(x) &= \answer{0}\\
\lim_{x\to0^+}f(x) &= \answer{-\infty}\\
\lim_{x\to0}f(x) &= \answer{DNE}\\
\lim_{x\to-\infty}f(x) &= \answer{\infty}\\
\lim_{x\to\infty}f(x) &= \answer{-6}
\end{align*}

\begin{exercise}

The only vertical asymptote of $f$ is $x=\answer{0}$.

\begin{exercise}

The only horizontal asymptote of $f$ is $y=\answer{-6}$.

\begin{exercise}

The intervals of continuity for $f$ from left to right are $\left(\answer{-\infty},\answer{0}\right]$, $\left(\answer{0},\answer{3}\right)$, $\left(\answer{3},\answer{5}\right]$, and $\left(\answer{5},\answer{\infty}\right)$.

\end{exercise}
\end{exercise}
\end{exercise}
\end{exercise}
