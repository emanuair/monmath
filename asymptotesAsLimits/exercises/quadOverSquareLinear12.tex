\begin{exercise}
Consider 
\[
f(x) = \frac{x^2+4 x+3}{(x-2)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{x\to 2^- } f(x) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{x\to 2^+ } f(x) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{x\to 2 } f(x) \begin{prompt} = \answer{\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
