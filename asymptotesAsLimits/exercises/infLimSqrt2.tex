\begin{exercise}
Let 
\[
g(v) = \frac{v^3-3 v^2-3 v+2}{4 v^2+\sqrt{v^6-4 v^3+3 v^2}+2 v-3}.
\]
Compute
\begin{enumerate}
\item $\lim_{v\to \infty} g(v) \begin{prompt} = \answer{1}\end{prompt}$
\item $\lim_{v\to -\infty}g(v) \begin{prompt} = \answer{-1}\end{prompt}$
\end{enumerate}
\begin{hint}
Multiply by
\[
\frac{\frac{1}{v^3}}{\frac{1}{v^3}}
\]
\end{hint}
\end{exercise}
