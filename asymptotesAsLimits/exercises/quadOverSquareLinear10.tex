\begin{exercise}
Consider 
\[
f(t) = \frac{t^2-4}{(t+4)^2}.
\]
Compute
\begin{enumerate}
\item $\lim_{t\to -4^- } f(t) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{t\to -4^+ } f(t) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{t\to -4 } f(t) \begin{prompt} = \answer{\infty}\end{prompt}$
\end{enumerate}
\end{exercise}
